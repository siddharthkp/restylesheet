### Install

```
npm install restylesheet --save
```

### Usage

```jsx
import React from 'react'
import Stylesheet from "./restylesheet"

const MyAwesomeApp = () => <div>
  <Stylesheet url="style.css" />
  ermahgerd!
</div>
```
